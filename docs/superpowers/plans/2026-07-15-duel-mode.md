# Duel Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ambient auto-battles: wild same-franchise opponents appear during work, fight the pet with a pre-computed battle script, deplete real HP, award EXP/dex entries on wins, and faint the pet on losses.

**Architecture:** `pet-core.sh` generates the entire fight up front into `$CACHE/duel.json` (atomic, seeded jq LCG); renderers replay it by wall clock; outcome is applied exactly once by the next resolve after `end_ts`. HP moves from a dead formula to a date-stamped file mutated by duels/heals. Spec: `docs/superpowers/specs/2026-07-15-duel-mode-design.md`.

**Tech Stack:** bash 3.2 + jq (core), JXA/CoreAnimation (overlay), Python ≥3.8 stdlib (terminal), existing test harness (`tests/lib.sh` seams + unittest + pty).

## Global Constraints

- bash 3.2 compatible: no `mapfile`, no `${var,,}`, no associative arrays
- the hook path (`pet-core.sh event …`) must always exit 0 silently
- Python ≥3.8 **stdlib only**; all state in `~/.cache/claude-pokemon-pet`; plugin dir never written at runtime
- atomic writes everywhere: `mktemp`+`mv` (bash), tmp+`os.replace` (python)
- jq LCG uses small constants (`(x*75+74) % 65537`) — bigger multipliers overflow IEEE-double exactness
- **FX hard rules (CLAUDE.md):** no flying projectiles of any color, ever; star-glint particles only (never circles); vpet attacks lunge with NO sparks; vpet tint warm white, never green
- Korean copy: use josa helpers (은/는, 이/가, 을/를, (으)로); unverified ko attack names fall back to 필살기; never mix languages in one locale
- JXA: no `$([...])` for NSArray; NSImage (not CGImageRef) into layer contents; probe NEW bridge patterns in isolated osascript first; FX wrapped in try/catch
- test seams: `PET_TODAY`, `PET_YESTERDAY`, `PET_NOW`, `PET_SEED`, `PET_LANG`, `PET_SHINY`; tests run under hermetic temp `$HOME` via `tests/lib.sh`
- conventional commits with colon separator; version bump to `0.11.0` in the final task

**Branch:** create `feat/duel-mode` from up-to-date `main` before Task 1 (`git checkout main && git pull && git checkout -b feat/duel-mode`).

**LCG cheat-sheet for tests** (`next = (seed*75+74) % 65537`): the wild-encounter roll fires when `next % 4 == 0`, i.e. when `seed % 4 == 2` (since `75s+74 ≡ 3s+2 (mod 4)`). Encounter seed = `partner.seed + tasks*7 + duels_today*13`, overridden entirely by `PET_SEED`. So `PET_SEED=2` **fires**, `PET_SEED=1` does **not**.

**3b.** After `heal_or_revive`, add the generator:

```bash
# ── duels: pre-computed battle scripts (spec 2026-07-15) ──
DUEL_JQ='
  def nxt: (. * 75 + 74) % 65537;
  ($pk[0]) as $pack | ($rs[0]) as $r |
  [foreach range(0; 24) as $_ ($seed | nxt; nxt; .)] as $rolls |
  ($pack.lines | length) as $nl |
  # stage-matched foe: same position in a random line; reroll once on self
  (if ($pack.edges // null) != null then
     def walk($sp; $n):
       if $n <= 0 then $sp else
         ((($pack.edges // {})[$sp] // []) | map(select(.quality != "reject"))) as $e |
         if ($e | length) == 0 then $sp else walk($e[0].to; $n - 1) end
       end;
     def foe_of($i): walk($pack.lines[$i].mons[0]; $r.stage - 1);
     (foe_of($rolls[0] % $nl)) as $try |
     (if $try == $r.species then foe_of($rolls[1] % $nl) else $try end)
   else
     def foe_of($i): ($pack.lines[$i].mons) as $m | $m[[($r.stage - 1), (($m | length) - 1)] | min];
     (foe_of($rolls[0] % $nl)) as $try |
     (if $try == $r.species then foe_of($rolls[1] % $nl) else $try end)
   end) as $foe |
  ($pack.lines[$rolls[0] % $nl].type // "normal") as $ftype |
  ($pack.species[$foe]) as $fs |
  (if $lang == "ko" then ($fs.names.ko // $fs.names.en) else $fs.names.en end) as $fname |
  species_move($pack; $foe; $lang; $ftype) as $fmove |
  species_elem($pack; $foe; $ftype) as $felem |
  ([($r.tasks + ($rolls[2] % 5) - 2), 1] | max) as $flevel |
  ($r.moves[0] // "ATTACK") as $pmove |
  # alternating damage rolls until a side drops; pet swings first, +4 bias
  def fight($php; $fhp; $i; $acc):
    if $php <= 0 or $fhp <= 0 or $i >= 20 then $acc
    else
      (if $i % 2 == 0 then "pet" else "foe" end) as $side |
      (18 + ($rolls[$i + 3] % 18) + (if $side == "pet" then 4 else 0 end)) as $dmg |
      (if $side == "pet" then [($fhp - $dmg), 0] | max else $fhp end) as $nf |
      (if $side == "foe" then [($php - $dmg), 0] | max else $php end) as $np |
      fight($np; $nf; $i + 1;
            $acc + [{t: (3 + 4 * $i), side: $side,
                     move: (if $side == "pet" then $pmove else $fmove end),
                     dmg: $dmg, pet_hp: $np, foe_hp: $nf}])
    end;
  fight($hp; 100; 0; []) as $turns |
  {
    date: $today, start_ts: $now,
    end_ts: ($now + $turns[-1].t + 4),
    kind: $kind,
    opponent: { species: $foe, name: $fname, level: $flevel,
                element: $felem, move: $fmove, franchise: $r.franchise },
    turns: $turns,
    result: (if $turns[-1].foe_hp == 0 then "win" else "lose" end),
    applied: false
  }'

duel_active() {
    [ -f "$CACHE/duel.json" ] || return 1
    local end ddate
    ddate="$(jq -r '.date // ""' "$CACHE/duel.json" 2>/dev/null)" || return 1
    end="$(jq -r '.end_ts // 0' "$CACHE/duel.json" 2>/dev/null)" || return 1
    [ "$ddate" = "$TODAY" ] && [ "$NOW" -lt $(( ${end:-0} + 6 )) ]
}

gen_duel() { # [force] — returns 0 iff a duel was generated
    command -v jq >/dev/null 2>&1 || return 1
    [ -f "$CACHE/resolved.json" ] || return 1
    duel_active && return 1
    [ "$(cur_state)" = "fainted" ] && return 1
    local pack duels seed tmp
    pack="$(pack_file "$(active_franchise)")"
    [ -f "$pack" ] || return 1
    duels="$(read_daily duels_today)"
    seed=$(( $(jq -r '.seed // 0' "$CACHE/partner" 2>/dev/null || echo 0) \
             + $(read_daily tasks) * 7 + duels * 13 ))
    [ -n "${PET_SEED:-}" ] && seed="$PET_SEED"
    if [ "${1:-}" != "force" ]; then
        [ "$duels" -lt 3 ] || return 1
        [ $(( (seed * 75 + 74) % 65537 % 4 )) -eq 0 ] || return 1
    fi
    tmp="$(mktemp)"
    jq -n --slurpfile pk "$pack" --slurpfile rs "$CACHE/resolved.json" \
       --argjson seed "$seed" --argjson hp "$(read_hp)" --argjson now "$NOW" \
       --arg today "$TODAY" --arg kind "$( [ "${1:-}" = force ] && echo manual || echo wild )" \
       --arg lang "$(cur_lang)" \
       "$JQ_DEFS$DUEL_JQ" > "$tmp" 2>/dev/null && mv "$tmp" "$CACHE/duel.json" || { rm -f "$tmp"; return 1; }
    [ "${1:-}" != "force" ] && bump_daily duels_today
    return 0
}

cmd_duel() {
    command -v jq >/dev/null 2>&1 || { echo "duel: jq required" >&2; exit 1; }
    [ -f "$CACHE/resolved.json" ] || cmd_resolve
    if [ "$(cur_state)" = "fainted" ]; then
        echo "your pet has fainted — complete a task to revive it"
        return 0
    fi
    if gen_duel force; then
        cmd_resolve
        echo "⚔ a wild $(jq -r '.opponent.name' "$CACHE/duel.json") appeared!"
    else
        echo "a duel is already underway"
    fi
}
```

```python
def duel_phase(d, now):
    """Where the wall clock sits inside a battle script."""
    if now < d["start_ts"] + 3:
        return "entrance", -1
    if now >= d["end_ts"] + 6:
        return "over", len(d["turns"]) - 1
    if now >= d["end_ts"]:
        return "result", len(d["turns"]) - 1
    ti = -1
    for i, turn in enumerate(d["turns"]):
        if now >= d["start_ts"] + turn["t"]:
            ti = i
    return ("entrance", -1) if ti < 0 else ("turn", ti)


def duel_caption(d, r, now):
    phase, ti = duel_phase(d, now)
    foe = d["opponent"]["name"]
    pet = r["name"]

def join_sprites(left, right, left_w, gap):
    """Two half-block sprite blocks side by side; left column stays a fixed
    left_w visible cells so the right sprite never wobbles."""
    rows = max(len(left), len(right))
    left = [""] * (rows - len(left)) + left      # pad short sprites on top
    right = [""] * (rows - len(right)) + right
    out = []
    for l, r_line in zip(left, right):
        pad = left_w - visible_len(l)
        out.append(l + " " * max(0, pad + gap) + r_line)
    return out
```
